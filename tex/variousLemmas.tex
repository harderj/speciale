
\begin{prop} For $x > 0$.
  \[ \int_x^\infty e^{-t^2/2} \dif t \leq \frac{1}{x} e^{-x^2/2} \]
\end{prop}

\begin{proof}
  Observe that for $t \geq x > 0$ we have $1 \leq t/x$ so
  \begin{align*}
    \int_x^\infty e^{-t^2/2} \dif t
    &\leq \int_x^\infty \frac{t}{x} e^{-t^2/2} \dif t
    \\ &\leq \frac{1}{x} e^{-x^2/2}
  \end{align*}
\end{proof}
